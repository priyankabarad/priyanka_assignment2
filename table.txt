CREATE TABLE books ( BookID INT AUTO_INCREMENT PRIMARY KEY,
 BookName VARCHAR(100) NOT NULL, 
 BookDescription VARCHAR(50) NOT NULL,
  QuantityAvailable VARCHAR(50) NOT NULL, 
  Price VARCHAR(50) NOT NULL, 
  AuthorName VARCHAR(100) NOT NULL, 
  Publisher VARCHAR(100) NOT NULL, 
  PublicationYear VARCHAR(50) NOT NULL,
 Genre VARCHAR(50) NOT NULL, 
   Language VARCHAR(50) NOT NULL,
    ProductAddedBy VARCHAR(50) NOT NULL );



CREATE TABLE admin ( id INT AUTO_INCREMENT PRIMARY KEY, username VARCHAR(255) NOT NULL UNIQUE, password VARCHAR(255) NOT NULL );
